**Changes to the detection block (currently lines 62-96):**

Replace the existing detection block with a three-way check. The new priority order is IDA > BN > Ghidra per CONTEXT.md locked decision.

New detection block:

```bash
# ---- Backend detection: IDA Pro > Binary Ninja > Ghidra ----
# Only one disassembler MCP server is registered per session.
# If the selected backend fails validation, we FAIL (no silent fallback).

mcp_name=""
mcp_type=""      # "stdio" or "sse"
mcp_command=""
mcp_args=""
mcp_url=""       # SSE URL (only for IDA)
mcp_env=""

if [ -d "/opt/ida-pro" ] && command -v idalib-mcp >/dev/null 2>&1; then
  # Validate idalib is properly activated before declaring IDA ready
  if ! python3 -c "import ida_idaapi" 2>/dev/null; then
    echo "[mcp] ERROR: IDA Pro is installed but idalib is not activated" >&2
    echo "[mcp] Expected 'import ida_idaapi' to succeed. Check py-activate-idalib.py ran during build." >&2
    exit 1
  fi
  mcp_name="ida_pro_mcp"
  mcp_type="sse"
  mcp_url="http://localhost:8745/sse"
  mcp_env="{\"IDADIR\": \"/opt/ida-pro\"}"
  echo "[mcp] Using IDA Pro (highest priority installed)"
elif [ -f /opt/binaryninja/scripts/install_api.py ] && [ -f "${BINJA_ROOT}/binary_ninja_headless_mcp.py" ]; then
  mcp_name="binary_ninja_headless_mcp"
  mcp_type="stdio"
  mcp_command="python3"
  mcp_args="[\"${BINJA_ROOT}/binary_ninja_headless_mcp.py\"]"
  mcp_env="{}"
  echo "[mcp] Using Binary Ninja (IDA Pro not installed)"
elif [ -f "${GHIDRA_ROOT}/ghidra_headless_mcp.py" ] || { [ -f "${GHIDRA_ROOT}/pyproject.toml" ] && [ -d "${GHIDRA_ROOT}/ghidra_headless_mcp" ]; }; then
  mcp_name="ghidra_headless_mcp"
  mcp_type="stdio"
  mcp_command="python3"
  mcp_args="[\"${GHIDRA_ROOT}/ghidra_headless_mcp.py\"]"
  ghidra_install_dir="${GHIDRA_INSTALL_DIR:-}"
  if [ -z "${ghidra_install_dir}" ] && [ -d /usr/share/ghidra ]; then
    ghidra_install_dir="/usr/share/ghidra"
  fi
  if [ -n "${ghidra_install_dir}" ]; then
    mcp_env="{\"GHIDRA_INSTALL_DIR\": \"${ghidra_install_dir}\"}"
  else
    mcp_env="{}"
  fi
  echo "[mcp] Using Ghidra (neither IDA Pro nor Binary Ninja installed)"
else
  echo "warning: no MCP backend (IDA Pro, Binary Ninja, or Ghidra) found, skipping MCP configuration" >&2
  printf '%s\n' '{"mcpServers":{}}' > "${CLAUDE_PROJECT_MCP}"
  write_codex_base_config
  exit 0
fi
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-02-01 | Spoofing | idalib-mcp SSE endpoint | accept | Phase 1 is localhost-only (container internal). SSE server on 127.0.0.1:8745 is not exposed outside the container. Remote exposure is Phase 2/3 scope with auth. |
| T-02-02 | Tampering | .mcp.json generation | accept | Script runs as part of container entrypoint with agent user permissions. Same trust model as existing BN/Ghidra config generation. |
